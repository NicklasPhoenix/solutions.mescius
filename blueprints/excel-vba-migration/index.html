// Initialize component with advanced configuration
function initializeComponent() {
    const config = {
        autoResize: true,
        enableVirtualization: true,
        theme: 'modern',
        dataSource: '/api/data',
        features: {
            sorting: true,
            filtering: true,
            editing: true
        }
    };
    
    const component = new MesciusComponent('#container', config);
    
    // Handle data updates
    component.onDataChanged = (data) => {
        console.log('Data updated:', data);
    };
    
    return component;
}
// Server-side implementation in C#
[ApiController]
[Route("api/[controller]")]
public class DataController : ControllerBase
{
    private readonly IDataService _dataService;
    
    public DataController(IDataService dataService)
    {
        _dataService = dataService;
    }
    
    [HttpGet]
    public async Task<IActionResult> GetData()
    {
        var data = await _dataService.GetDataAsync();
        return Ok(data);
    }
    
    [HttpPost]
    public async Task<IActionResult> UpdateData([FromBody] DataModel model)
    {
        await _dataService.UpdateAsync(model);
        return Ok();
    }
}
