// React/TypeScript integration - Adastra's target platform
import React, { useEffect, useRef, useState } from 'react';
import * as wijmo from 'wijmo/wijmo';
import { FlexChart } from 'wijmo/wijmo.react.chart';

interface ChartData {
  timestamp: Date;
  revenue: number;
  profit: number;
  expenses: number;
}

const BusinessChart: React.FC = () => {
  const [data, setData] = useState<ChartData[]>([]);
  const [loading, setLoading] = useState(true);
  
  useEffect(() => {
    // Load initial data
    fetchBusinessData()
      .then((chartData: ChartData[]) => {
        setData(chartData);
        setLoading(false);
      })
      .catch(console.error);
  }, []);

  return (
    <div className="chart-container">
      {loading ? (
        <div className="loading">Loading chart data...</div>
      ) : (
        <FlexChart
          itemsSource={data}
          bindingX="timestamp"
          chartType={wijmo.chart.ChartType.Line}
          legend={wijmo.chart.Position.Bottom}
        >
          <wijmo.chart.Series binding="revenue" name="Revenue" />
          <wijmo.chart.Series binding="profit" name="Profit" />
          <wijmo.chart.Series binding="expenses" name="Expenses" />
        </FlexChart>
      )}
    </div>
  );
};

// Real-time chart updates - Live BI dashboard functionality
class RealTimeChartManager {
    constructor(chart, updateInterval = 5000) {
        this.chart = chart;
        this.updateInterval = updateInterval;
        this.isRunning = false;
        this.intervalId = null;
    }
    
    start() {
        if (this.isRunning) return;
        
        this.isRunning = true;
        this.intervalId = setInterval(() => {
            this.updateChartData();
        }, this.updateInterval);
        
        console.log('Real-time updates started');
    }
    
    stop() {
        if (this.intervalId) {
            clearInterval(this.intervalId);
            this.intervalId = null;
        }
        this.isRunning = false;
        console.log('Real-time updates stopped');
    }
    
    async updateChartData() {
        try {
            // Fetch latest data from API
            const newData = await fetchLatestBusinessMetrics();
            
            // Get current data
            const currentData = this.chart.itemsSource || [];
            
            // Maintain rolling window of last 100 data points
            const maxPoints = 100;
            const updatedData = [...currentData, ...newData]
                .slice(-maxPoints)
                .sort((a, b) => new Date(a.timestamp) - new Date(b.timestamp));
            
            // Update chart with smooth transition
            this.chart.beginUpdate();
            this.chart.itemsSource = updatedData;
            this.chart.endUpdate();
            
            // Auto-scale axes to show recent data
            this.autoScaleAxes(updatedData);
            
        } catch (error) {
            console.error('Failed to update chart data:', error);
        }
    }
    
    autoScaleAxes(data) {
        if (!data || data.length === 0) return;
        
        // Show last 24 hours of data
        const now = new Date();
        const yesterday = new Date(now.getTime() - 24 * 60 * 60 * 1000);
        
        this.chart.axisX.min = yesterday;
        this.chart.axisX.max = now;
    }
}

// Initialize real-time updates
const realtimeManager = new RealTimeChartManager(flexChart, 5000);
realtimeManager.start();
