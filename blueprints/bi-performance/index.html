// Real-time updates for BI dashboards
function startRealTimeUpdates() {
    setInterval(() => {
        // Update random records every 2 seconds
        const recordsToUpdate = Math.min(10, sampleData.length);
        
        for (let i = 0; i < recordsToUpdate; i++) {
            const randomIndex = Math.floor(Math.random() * sampleData.length);
            const record = sampleData[randomIndex];
            
            // Simulate live business data changes
            record.revenue += (Math.random() - 0.5) * 100000;
            record.growth = (Math.random() - 0.5) * 0.5;
            record.lastUpdated = new Date();
        }
        
        // Refresh grid to show changes
        flexGrid.collectionView.refresh();
    }, 2000);
}

// CSV export for reports
function exportToCSV() {
    const items = flexGrid.collectionView.items;
    let csv = 'ID,Company,Revenue,Growth\n';
    
    items.forEach(item => {
        csv += `${item.id},"${item.companyName}",${item.revenue},${item.growth}\n`;
    });
    
    // Download file
    const blob = new Blob([csv], { type: 'text/csv' });
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url;
    a.download = 'bi-data.csv';
    a.click();
}
